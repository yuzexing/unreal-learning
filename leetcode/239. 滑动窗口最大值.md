# 239. 滑动窗口最大值

### 初次提交
```
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        if (nums.size() - k  + 1 < 0) {
            int max = INT_MIN;
            for (int n : nums) {
                max = std::max(max, n);
            }
            return vector<int>{max};
        }
        int max = INT_MIN;
        std::deque<int> dq;
        vector<int> res;
        res.reserve(nums.size() - k  + 1);
        for (int i = 0; i < k; i++) {
            dq.push_back(nums[i]);
        }
        for (auto it = dq.begin(); it != dq.end(); it ++) {
            max = std::max(max, *it);
        }
        res.push_back(max);
        for (int i = k; i < nums.size(); i ++) {
            dq.pop_front();
            dq.push_back(nums[i]);
            int max = INT_MIN;
            for (auto it = dq.begin(); it != dq.end(); it ++) {
                max = std::max(max, *it);
            }
            res.push_back(max);
        }
        return res;
    }
};
```
