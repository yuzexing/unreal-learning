# 151. 反转字符串中的单词

### 初次提交
```
class Solution {
public:
    string reverseWords(string s) {
        std::stack<std::string_view> sk;
        std::string_view sv(s.c_str());
        int i = 0;
        int j = 0;
        while (j < s.size()) {
            while (s[i] == ' ' && i < s.size()) {
                i++;
            }
            if (sv[j] == ' ' && i < j) {
                std::string_view sv_temp = sv.substr(i, j - i);
                sk.push(sv_temp);
                j++;
                i = j;
            }
            else {
                j++;
            }
        }
        if (i < j) {
            std::string_view sv_temp = sv.substr(i, j - i);
            sk.push(sv_temp);
        }
        string res;
        res.reserve(s.size());
        while (!sk.empty()) {
            string_view sv_temp = sk.top();
            res += sv_temp;
            sk.pop();
            if (!sk.empty()) {
                res += ' ';
            }
        }
        return res;
    }
};
```
