# 208. 实现 Trie (前缀树)

### 初次提交
```
class Trie {
    struct TreeNode {
        char c;
        bool end = false;
        unordered_map<char, TreeNode*> m;
    };
private:
    unordered_map<char, TreeNode*> tree;
public:
    Trie() {
        
    }

    void insert(const string& word) {
        unordered_map<char, TreeNode*>* tempTree = &tree;
        TreeNode* targetNode = nullptr;
        for (char c : word) {
            auto it = tempTree->find(c);
            if (it != tempTree->end()) {
                targetNode = it->second;
            }
            else {
                TreeNode* node = new TreeNode();
                node->c = c;
                node->end = false;
                tempTree->insert({ c, node });
                targetNode = node;
            }
            tempTree = &targetNode->m;
        }
        targetNode->end = true;
    }

    bool search(const string& word) {
        auto tempTree = &tree;
        TreeNode* targetNode = nullptr;
        for (char c : word) {
            auto it = tempTree->find(c);
            if (it != tempTree->end()) {
                targetNode = it->second;
                tempTree = &targetNode->m;
                continue;
            }
            return false;
        }
        return targetNode && targetNode->end;
    }

    bool startsWith(const string& prefix) {
        auto tempTree = &tree;
        TreeNode* targetNode = nullptr;
        for (char c : prefix) {
            auto it = tempTree->find(c);
            if (it != tempTree->end()) {
                targetNode = it->second;
                tempTree = &targetNode->m;
                continue;
            }
            return false;
        }
        return true;
    }
};
```
