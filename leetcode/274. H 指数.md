# 274. H 指数


### 初次提交
```
class Solution {
public:
    int hIndex(vector<int>& citations) {
        std::sort(citations.begin(), citations.end());
        int num = 0;
        int maxNum = -1;
        for (int i = citations.size() - 1; i >= 0; i --) {
            num ++;
            maxNum = std::max(maxNum, num >= citations[i] ? citations[i] : num);
        }
        return maxNum;
    }
};
```
