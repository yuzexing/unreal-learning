# 238. 除自身以外数组的乘积

### 初次提交
```
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        vector<int> res(nums.size(), 1);
        int preTotal = 1;
        for (int i = 0; i < nums.size(); i ++) {
            res[i] *= preTotal;
            preTotal *= nums[i];
        }
        preTotal = 1;
        for (int i = nums.size() - 1; i >= 0; i --) {
            res[i] *= preTotal;
            preTotal *= nums[i];
        }
        return res;
    }
};
```
