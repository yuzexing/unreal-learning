# 120. 三角形最小路径和

### 初次提交
```
class Solution {
public:
    int minimumTotal(vector<vector<int>>& triangle) {
        // 将三角形从下向上遍历，选取小的加上去
        for (int i = triangle.size() - 1; i >= 1; i --) {
            for (int j = 0; j < triangle[i].size() - 1; j ++) {
                if (triangle[i][j] < triangle[i][j + 1]) {
                    triangle[i - 1][j] += triangle[i][j];
                } else {
                    triangle[i - 1][j] += triangle[i][j + 1];
                }
            }
        }
        return triangle[0][0];
    }
};
```
