# 148. 排序链表

### 初次提交
```
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        vector<int> v;
        ListNode* node = head;
        while (node) {
            v.push_back(node->val);
            node = node->next;
        }
        node = head;
        std::sort(v.begin(), v.end());
        int i = 0;
        while (node) {
            node->val = v[i];
            node = node->next;
            i++;
        }
        return head;
    }
};
```
