# 41. 缺失的第一个正数

### 初次提交
```
class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        std::sort(nums.begin(), nums.end());
        int begin = 1;
        for (int v : nums) {
            if (v < begin) {
                continue;
            } else if (v == begin) {
                begin ++;
            } else {
                return begin;
            }
        }
        return begin;
    }
};
```
